To get a random JSON via api, use the following methods:

  1. POST: http://utility-app.herokuapp.com/api/randomJson
    Headers: 
    {"Content-Type" : "application/json"}
    Body: 
    {schema : { ... a valid schema ...} }
  2. GET: http://utility-app.herokuapp.com/api/randomJson?schemaURL=[Link to a valid schema]
This is an example of a valid schema
{
	"integer" : "INTEGER:MAX:MIN",
	"double" : "DOUBLE:PRECISION:MAX:MIN",
	"url" : "URL:PROTOCOL",
	"currency" : "CURRENCY",
	"textarea" : "TEXTAREA",
	"textarea_words" : "TEXTAREA:true",
	"string":"STRING", 
	"date":"DATE", 
	"boolean":"BOOLEAN",
	"email" : "EMAIL",
	"subnode" : {
		"string":"STRING", 
		"date":"DATE", 
		"boolean":"BOOLEAN",
		"integer" : "INTEGER",
		"double" : "DOUBLE",
		"url" : "URL",
		"array_subnode" : [
			{"url": "URL"}
		]
	},
	"array_objects" : [
		{
			"string" : "STRING",
			"date" : "DATE"
		}
	],
	"array_values" : ["INTEGER"]
}
